25 /// Generic collection of elements that are roughly contiguous in memory.
26 ///
27 /// The most notable feature of the `rerun::Collection` is that its data may be either **owned** or **borrowed**:
28 /// * Borrowed: If data is borrowed it *must* outlive its source (in particular, the pointer to
29 /// the source mustn't invalidate)
30 /// * Owned: Owned data is copied into an internal std::vector
31 ///
32 /// Collections are either filled explicitly using `Collection::borrow` &`Collection::take_ownership`
33 /// or (most commonly in user code) implicitly using the `CollectionAdapter` trait
34 /// (see documentation for `CollectionAdapter` for more information on how data can be adapted).
35 ///
36 /// Other than being assignable, collections are generally immutable:
37 /// there is no mutable data access in order to not violate the contract with the data lender
38 /// and changes in size are not possible.
39 ///
40 /// ## Implementation notes:
41 ///
42 /// Does intentionally not implement copy construction since this for the owned case this may
43 /// be expensive. Typically, there should be no need to copy rerun collections, so this more
44 /// than likely indicates a bug inside the Rerun SDK.
45 template <typename TElement>
46 class Collection {
47 public:
48 /// Type of the elements in the collection.
49 ///
50 /// Note that calling this `value_type` makes it compatible with the STL.
51 using value_type = TElement;
52
53 /// Type of an adapter given an input container type.
54 ///
55 /// Note that the "container" passed may also be a single element of something.
56 /// The only thing relevant is that there's an Adapter for it.
57 template <typename TContainer>
58 using Adapter = CollectionAdapter<
59 TElement, std::remove_cv_t<std::remove_reference_t<TContainer>>,
60 std::enable_if_t<true>>;
61
62 /// Creates a new empty collection.
63 Collection() : ownership(CollectionOwnership::Borrowed) {
64 storage.borrowed.data = nullptr;
65 storage.borrowed.num_instances = 0;
66 }
67
68 /// Construct using a `CollectionAdapter` for the given input type.
69 template <
70 typename TContainer, //
71 // Avoid conflicting with the copy/move constructor.
72 // We could implement this also with an adapter, but this might confuse trait checks like `std::is_copy_constructible`.
73 typename = std::enable_if_t<
74 !std::is_same_v<std::remove_reference_t<TContainer>, Collection<TElement>>> //
75 >
76 Collection(TContainer&& input)
77 : Collection(Adapter<TContainer>()(std::forward<TContainer>(input))) {}

79 /// Copy constructor.
80 ///
81 /// If the data is owned, this will copy the data.
82 /// If the data is borrowed, this will copy the borrow,
83 /// meaning there's now (at least) two collections borrowing the same data.
84 Collection(const Collection<TElement>& other) : ownership(other.ownership) {
85 switch (other.ownership) {
86 case CollectionOwnership::Borrowed: {
87 storage.borrowed = other.storage.borrowed;
88 break;
89 }
90
91 case CollectionOwnership::VectorOwned: {
92 new (&storage.vector_owned) std::vector<TElement>(other.storage.vector_owned);
93 break;
94 }
95 }
96 }

98 /// Copy assignment.
99 ///
100 /// If the data is owned, this will copy the data.
101 /// If the data is borrowed, this will copy the borrow,
102 /// meaning there's now (at least) two collections borrowing the same data.
103 void operator=(const Collection<TElement>& other) {
104 this->~Collection<TElement>();
105 new (this) Collection(other);
106 }

108 /// Move constructor.
109 Collection(Collection<TElement>&& other) : Collection() {
110 swap(other);
111 }

113 /// Move assignment.
114 void operator=(Collection<TElement>&& other) {
115 // Need to disable the maybe-uninitialized here. It seems like the compiler may be confused in situations where
116 // we are assigning into an unused optional from a temporary. The fact that this hits the move-assignment without
117 // having called the move constructor is suspicious though and hints of an actual bug.
118 //

123 /// Construct from a initializer list of elements that are compatible with TElement.
124 ///
125 /// Takes ownership of the passed elements.
126 /// If you want to avoid an allocation, you have to manually keep the data on the stack
127 /// (e.g. as `std::array`) and construct the collection from this instead.
128 ///
129 /// This is not done as a `CollectionAdapter` since it tends to cause deduction issues
130 /// (since there's special rules for overload resolution for initializer lists)
131 Collection(std::initializer_list<TElement> data)
132 : ownership(CollectionOwnership::VectorOwned) {
133 // Don't assign, since the vector is in an undefined state and assigning may
134 // attempt to free data.
135 new (&storage.vector_owned) std::vector<TElement>(data);
136 }
137
138 /// Borrows binary compatible data into the collection.
139 ///
140 /// Borrowed data must outlive the collection!
141 /// (If the pointer passed is into an std::vector or similar, this std::vector mustn't be
142 /// resized.)
143 /// The passed type must be binary compatible with the collection type.
144 ///
145 /// Since `rerun::Collection` does not provide write access, data is guaranteed to be unchanged by
146 /// any function or operation taking on a `Collection`.
147 template <typename T>
148 static Collection<TElement> borrow(const T* data, size_t num_instances) {
149 static_assert(
150 sizeof(T) == sizeof(TElement),
151 "T & TElement are not binary compatible: Size mismatch."
152 );
153 static_assert(
154 alignof(T) <= alignof(TElement),
155 "T & TElement are not binary compatible: TElement has a higher alignment requirement than T. This implies that pointers to T may not have the alignment needed to access TElement."
156 );
157
158 Collection<TElement> batch;
159 batch.ownership = CollectionOwnership::Borrowed;
160 batch.storage.borrowed.data = reinterpret_cast<const TElement*>(data);
161 batch.storage.borrowed.num_instances = num_instances;
162 return batch;
163 }
164
165 /// Borrows binary compatible data into the collection.
166 ///
167 /// Version of `borrow` that takes a void pointer, omitting any checks.
168 ///
169 /// Borrowed data must outlive the collection!
170 /// (If the pointer passed is into an std::vector or similar, this std::vector mustn't be
171 /// resized.)
172 ///
173 /// Since `rerun::Collection` does not provide write access, data is guaranteed to be unchanged by
174 /// any function or operation taking on a `rerun::Collection`.
175 static Collection borrow(const void* data, size_t num_instances) {
176 return borrow(reinterpret_cast<const TElement*>(data), num_instances);
177 }

179 /// Takes ownership of a temporary `std::vector`, moving it into the collection.
180 ///
181 /// Takes ownership of the data and moves it into the collection.
182 static Collection<TElement> take_ownership(std::vector<TElement>&& data) {
183 Collection<TElement> batch;
184 batch.ownership = CollectionOwnership::VectorOwned;
185 // Don't assign, since the vector is in an undefined state and assigning may
186 // attempt to free data.
187 new (&batch.storage.vector_owned) std::vector<TElement>(std::move(data));
188
189 return batch;
190 }
191
192 /// Takes ownership of a single element, moving it into the collection.
193 static Collection<TElement> take_ownership(TElement&& data) {
194 // TODO(andreas): there should be a special path here to avoid allocating a vector.
195 std::vector<TElement> elements;
196 elements.emplace_back(std::move(data));
197 return take_ownership(std::move(elements));
198 }

200 /// Swaps the content of this collection with another.
201 void swap(Collection<TElement>& other) {
202 // (writing out this-> here to make it less confusing!)
203 switch (this->ownership) {
204 case CollectionOwnership::Borrowed: {
205 switch (other.ownership) {
206 case CollectionOwnership::Borrowed:
207 std::swap(this->storage.borrowed, other.storage.borrowed);
208 break;
209
210 case CollectionOwnership::VectorOwned: {
211 auto this_borrowed_data_old = this->storage.borrowed;
212 new (&this->storage.vector_owned)
213 std::vector<TElement>(std::move(other.storage.vector_owned));
214 other.storage.borrowed = this_borrowed_data_old;
215 break;
216 }
217 }
218 break;
219 }
220
221 case CollectionOwnership::VectorOwned: {
222 switch (other.ownership) {
223 case CollectionOwnership::Borrowed: {
224 auto other_borrowed_data_old = other.storage.borrowed;
225 new (&other.storage.vector_owned)
226 std::vector<TElement>(std::move(this->storage.vector_owned));
227 this->storage.borrowed = other_borrowed_data_old;
228 break;
229 }
230
231 case CollectionOwnership::VectorOwned:
232 std::swap(storage.vector_owned, other.storage.vector_owned);
233 break;
234 }
235 break;
236 }
237 }
238
239 std::swap(ownership, other.ownership);
240 }

242 ~Collection() {
243 switch (ownership) {
244 case CollectionOwnership::Borrowed:
245 break; // nothing to do.
246 case CollectionOwnership::VectorOwned:
247 storage.vector_owned.~vector(); // Deallocate the vector!
248 break;
249 }
250 }

252 /// Returns the number of instances in this collection.
253 size_t size() const {
254 switch (ownership) {
255 case CollectionOwnership::Borrowed:
256 return storage.borrowed.num_instances;
257 case CollectionOwnership::VectorOwned:
258 return storage.vector_owned.size();
259 }
260 return 0;
261 }

263 /// Returns true if the collection is empty.
264 bool empty() const {
265 switch (ownership) {
266 case CollectionOwnership::Borrowed:
267 return storage.borrowed.num_instances == 0;
268 case CollectionOwnership::VectorOwned:
269 return storage.vector_owned.empty();
270 }
271 return 0;
272 }

274 /// Returns a raw pointer to the underlying data.
275 ///
276 /// Do not use this if the data is not continuous in memory!
277 /// TODO(#4257): So far it always is continuous, but in the future we want to support strides!
278 ///
279 /// The pointer is only valid as long as backing storage is alive
280 /// which is either until the collection is destroyed the borrowed source is destroyed/moved.
281 const TElement* data() const {
282 switch (ownership) {
283 case CollectionOwnership::Borrowed:
284 return storage.borrowed.data;
285 case CollectionOwnership::VectorOwned:
286 return storage.vector_owned.data();
287 }
