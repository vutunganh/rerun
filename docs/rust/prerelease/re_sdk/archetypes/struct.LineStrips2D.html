Expand description

Archetype: A batch of line strips with positions and optional colors, radii, labels, etc.

Example

line_strip2d_batch:

ⓘ
//! Log a batch of 2d line strips.

fn main() -> Result<(), Box<dyn std::error::Error>> {
    let (rec, storage) =
        rerun::RecordingStreamBuilder::new("rerun_example_line_strip2d").memory()?;

    let strip1 = [[0., 0.], [2., 1.], [4., -1.], [6., 0.]];
    #[rustfmt::skip]
    let strip2 = [[0., 3.], [1., 4.], [2., 2.], [3., 4.], [4., 2.], [5., 4.], [6., 3.]];
    rec.log(
        "strips",
        &rerun::LineStrips2D::new([strip1.to_vec(), strip2.to_vec()])
            .with_colors([0xFF0000FF, 0x00FF00FF])
            .with_radii([0.025, 0.005])
            .with_labels(["one strip here", "and one strip there"]),
    )?;

    // Log an extra rect to set the view bounds
    rec.log(
        "bounds",
        &rerun::Boxes2D::from_centers_and_sizes([(3.0, 1.5)], [(8.0, 9.0)]),
    )?;

    rerun::native_viewer::show(storage.take())?;
    Ok(())
}

Fields§

§strips: Vec<LineStrip2D, Global>

All the actual 2D line strips that make up the batch.

§radii: Option<Vec<Radius, Global>>

Optional radii for the line strips.

§colors: Option<Vec<Color, Global>>

Optional colors for the line strips.

§labels: Option<Vec<Text, Global>>

Optional text labels for the line strips.

§draw_order: Option<DrawOrder>

An optional floating point value that specifies the 2D drawing order of each line strip.

Objects with higher values are drawn on top of those with lower values.

§class_ids: Option<Vec<ClassId, Global>>

Optional ClassIds for the lines.

The class ID provides colors and labels if not specified explicitly.

§instance_keys: Option<Vec<InstanceKey, Global>>

Unique identifiers for each individual line strip in the batch.
